##### 并查集
```
  var fa []int
  func find(x int) int {
      if fa[x] == x {
          return x
      }
      
      fa[x] = find(fa[x])
      return fa[x]
  }

  func union(x int, y int) {
      fa[find(x)] = find(y)
  }
```
